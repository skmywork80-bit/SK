import 'package:flutter/material.dart'; void main() { runApp(MyApp()); } class MyApp extends StatelessWidget { @override Widget build(BuildContext context) { return MaterialApp( debugShowCheckedModeBanner: false, title: 'Word Learning App', theme: ThemeData(primarySwatch: Colors.blue), home: WordPage(), ); } } // --- প্রথম পেজ: শব্দের তালিকা দেখাবে --- class WordPage extends StatelessWidget { final List> words = [ { 'word': 'Look', 'pronunciation': 'লুক', 'meaning': 'দেখা', 'example': "Don't look at me (আমার দিকে তাকাবে না)" }, { 'word': 'Go', 'pronunciation': 'গো', 'meaning': 'যাওয়া', 'example': "Go to school (স্কুলে যাও)" }, // আরও শব্দ এখানে যুক্ত করতে পারো ]; @override Widget build(BuildContext context) { return Scaffold( appBar: AppBar(title: Text('Word List')), body: ListView.builder( itemCount: words.length, itemBuilder: (context, index) { var word = words[index]; return Card( margin: EdgeInsets.all(8), child: ListTile( title: Text('${word['word']} (${word['pronunciation']})'), subtitle: Column( crossAxisAlignment: CrossAxisAlignment.start, children: [ Text('Meaning: ${word['meaning']}'), Text('Example: ${word['example']}'), ], ), trailing: IconButton( icon: Icon(Icons.push_pin), onPressed: () { // পিনে ক্লিক → পাসওয়ার্ড চাওয়া হবে showDialog( context: context, builder: (context) => PasswordDialog(), ); }, ), ), ); }, ), ); } } // --- পাসওয়ার্ড ডায়ালগ --- class PasswordDialog extends StatefulWidget { @override _PasswordDialogState createState() => _PasswordDialogState(); } class _PasswordDialogState extends State { final _controller = TextEditingController(); final String correctPassword = '1234'; // ডেমো পাসওয়ার্ড @override Widget build(BuildContext context) { return AlertDialog( title: Text('Enter Password'), content: TextField( controller: _controller, obscureText: true, decoration: InputDecoration(hintText: 'Password'), ), actions: [ TextButton( child: Text('Cancel'), onPressed: () => Navigator.pop(context), ), TextButton( child: Text('Submit'), onPressed: () { if (_controller.text == correctPassword) { Navigator.pop(context); Navigator.push( context, MaterialPageRoute(builder: (context) => AdminOptionsPage()), ); } else { ScaffoldMessenger.of(context).showSnackBar( SnackBar(content: Text('Incorrect Password')), ); } }, ), ], ); } } // --- অ্যাডমিন অপশন পেজ --- class AdminOptionsPage extends StatelessWidget { @override Widget build(BuildContext context) { return Scaffold( appBar: AppBar(title: Text('Admin Options')), body: Padding( padding: EdgeInsets.all(16), child: Column( mainAxisAlignment: MainAxisAlignment.center, children: [ ElevatedButton( child: Text('Edit App'), onPressed: () { // Edit App functionality এখানে যুক্ত হবে }, ), SizedBox(height: 20), ElevatedButton( child: Text('My Accounts'), onPressed: () { // My Accounts functionality এখানে যুক্ত হবে }, ), ], ), ), ); } }
